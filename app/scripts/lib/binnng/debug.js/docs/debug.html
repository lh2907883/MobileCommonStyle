        render = (msg) ->
      text = ""
      arr = []

      if isArray msg
        arr.push "#{item}" for item in msg
        text = "[" + arr.join(',') + "]"

      else if isObejct msg
        arr.push "#{item}: #{msg[item]}" for item of msg
        text = "{" + arr.join(',') + "}"

      else
        text = String(msg)

      text
